```typescript
/**
 * FinanceReadFacade — Centralized read-only access to finance tables.
 *
 * PURPOSE:
 * Compliance (DSAR, retention), registration, and other modules need to read
 * finance data (invoices, payments, refunds, etc.) without coupling to the
 * finance module's internal schema. This facade provides a single, well-typed
 * entry point for all cross-module finance reads.
 *
 * CONVENTIONS:
 * - Every method starts with `tenantId: string` as the first parameter.
 * - No RLS transaction needed for reads — `tenant_id` is in every `where` clause.
 * - Returns `null` when a record is not found (callers decide whether to throw).
 * - Batch methods return arrays (empty array = nothing found).
 */
import { Injectable } from '@nestjs/common';

import { PrismaService } from '../prisma/prisma.service';

// ─── Common select shapes ─────────────────────────────────────────────────────

const INVOICE_SUMMARY_SELECT = {
  id: true,
  invoice_number: true,
  household_id: true,
  student_id: true,
  total_amount: true,
  balance_amount: true,
  status: true,
  due_date: true,
  issued_date: true,
  description: true,
  created_at: true,
} as const;

const PAYMENT_SUMMARY_SELECT = {
  id: true,
  payment_number: true,
  household_id: true,
  amount: true,
  payment_method: true,
  payment_date: true,
  status: true,
  notes: true,
  created_at: true,
} as const;

const REFUND_SUMMARY_SELECT = {
  id: true,
  refund_number: true,
  amount: true,
  reason: true,
  status: true,
  created_at: true,
  payment: {
    select: { id: true, household_id: true, payment_number: true },
  },
} as const;

const CREDIT_NOTE_SUMMARY_SELECT = {
  id: true,
  credit_note_number: true,
  household_id: true,
  amount: true,
  reason: true,
  status: true,
  created_at: true,
} as const;

const FEE_STRUCTURE_SUMMARY_SELECT = {
  id: true,
  name: true,
  amount: true,
  billing_frequency: true,
  year_group_id: true,
  status: true,
  created_at: true,
} as const;

const DISCOUNT_SUMMARY_SELECT = {
  id: true,
  name: true,
  discount_type: true,
  value: true,
  status: true,
  created_at: true,
} as const;

// ─── Facade ───────────────────────────────────────────────────────────────────

@Injectable()
export class FinanceReadFacade {
  constructor(private readonly prisma: PrismaService) {}

  // ─── Invoices ─────────────────────────────────────────────────────────────

  async findInvoicesByHousehold(tenantId: string, householdId: string) {
    return this.prisma.invoice.findMany({
      where: { tenant_id: tenantId, household_id: householdId },
      select: INVOICE_SUMMARY_SELECT,
      orderBy: { created_at: 'desc' },
    });
  }

  async countInvoicesBeforeDate(tenantId: string, cutoffDate: Date): Promise<number> {
    return this.prisma.invoice.count({
      where: { tenant_id: tenantId, created_at: { lt: cutoffDate } },
    });
  }

  // ─── Payments ─────────────────────────────────────────────────────────────

  async findPaymentsByHousehold(tenantId: string, householdId: string) {
    return this.prisma.payment.findMany({
      where: { tenant_id: tenantId, household_id: householdId },
      select: PAYMENT_SUMMARY_SELECT,
      orderBy: { created_at: 'desc' },
    });
  }

  // ─── Refunds ──────────────────────────────────────────────────────────────

  async findRefundsByHousehold(tenantId: string, householdId: string) {
    return this.prisma.refund.findMany({
      where: { tenant_id: tenantId, payment: { household_id: householdId } },
      select: REFUND_SUMMARY_SELECT,
      orderBy: { created_at: 'desc' },
    });
  }

  // ─── Credit Notes ─────────────────────────────────────────────────────────

  async findCreditNotesByHousehold(tenantId: string, householdId: string) {
    return this.prisma.creditNote.findMany({
      where: { tenant_id: tenantId, household_id: householdId },
      select: CREDIT_NOTE_SUMMARY_SELECT,
      orderBy: { created_at: 'desc' },
    });
  }

  // ─── Payment Plans ────────────────────────────────────────────────────────

  async findPaymentPlanRequestsByHousehold(tenantId: string, householdId: string) {
    return this.prisma.paymentPlanRequest.findMany({
      where: { tenant_id: tenantId, household_id: householdId },
      orderBy: { created_at: 'desc' },
    });
  }

  // ─── Scholarships ────────────────────────────────────────────────────────

  async findScholarshipsByStudent(tenantId: string, studentId: string) {
    return this.prisma.scholarship.findMany({
      where: { tenant_id: tenantId, student_id: studentId },
      orderBy: { created_at: 'desc' },
    });
  }

  // ─── Fee Structures ───────────────────────────────────────────────────────

  async findActiveFeeStructures(tenantId: string, yearGroupId?: string) {
    return this.prisma.feeStructure.findMany({
      where: {
        tenant_id: tenantId,
        status: 'active',
        ...(yearGroupId && { year_group_id: yearGroupId }),
      },
      select: FEE_STRUCTURE_SUMMARY_SELECT,
      orderBy: { name: 'asc' },
    });
  }

  // ─── Discounts ────────────────────────────────────────────────────────────

  async findActiveDiscounts(tenantId: string) {
    return this.prisma.discount.findMany({
      where: { tenant_id: tenantId, status: 'active' },
      select: DISCOUNT_SUMMARY_SELECT,
      orderBy: { name: 'asc' },
    });
  }
}
```
